query

5. python program to reverse given number
num = 12345
rev = 0

while num > 0:
    rev = rev * 10 + num % 10
    num //= 10

print("reversed number:", rev)

8. eliza chatbot (simple relational knowledge – python)
while True:
    user = input("you: ").lower()

    if user == "hi":
        print("eliza: hello")
    elif "sad" in user:
        print("eliza: why are you sad")
    elif "happy" in user:
        print("eliza: that's good to hear")
    elif user == "bye":
        print("eliza: goodbye")
        break
    else:
        print("eliza: tell me more")

9. simple voice bot (nlp + relational knowledge – python)
import pyttsx3

engine = pyttsx3.init()

while True:
    text = input("you: ").lower()

    if "hello" in text:
        engine.say("hello, how can i help you")
    elif "name" in text:
        engine.say("i am an ai voice bot")
    elif "bye" in text:
        engine.say("goodbye")
        engine.runAndWait()
        break
    else:
        engine.say("please say again")

    engine.runAndWait()
